export function SmoothScroll({ scrollTo, children, ...props }) {
  return (
    <button
      {...props}
      onClick={() =>
        document
          .getElementById(scrollTo)
          ?.scrollIntoView({ behavior: "smooth" })
      }
    >
      {children}
    </button>
  );
}
